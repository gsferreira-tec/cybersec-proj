\section{Introduction}
\begin{frame}{Why Email Authentication Matters}
\begin{itemize}
  \gitem Email remains a primary vector for phishing attacks.
  \vspace{0.5cm}
  \gitem SPF, DKIM, and DMARC were designed to mitigate spoofing.
  \vspace{0.5cm}  
  \gitem Misconfigurations still allow successful impersonation.
\end{itemize}
\end{frame}

\section{Lab Setup}
\begin{frame}[fragile]{Docker-Compose Architecture}
\begin{minted}{yaml}
version: "3"
services:
  dns-server:        # BIND
  victim-mail:       # Postfix/Dovecot
  attacker-sender:   # Espoofer + custom scripts
  webmail:           # User interface for the email service
\end{minted}
\vfill
\begin{itemize}
  \gitem DNS server exposes misconfigured SPF, DKIM, and DMARC records.
  \vspace{0.5cm}  
  \gitem Victim mail server runs on 10.9.0.6 with docker-mailserver.
  \vspace{0.5cm}
  \gitem Attacker container mounts Espoofer and custom attack scripts.
\end{itemize}
\end{frame}

\section{Misconfigurations Explored}
\begin{frame}{SPF / DKIM / DMARC Misconfigurations}
\begin{itemize}
  \gitem SPF: \texttt{v=spf1 a mx \textasciitilde all} versus \texttt{-all}.
  \vspace{0.5cm}  
  \gitem DKIM: weak selector, missing key rotation, or absent selector.
  \vspace{0.5cm}  
  \gitem DMARC: \texttt{p=none} versus \texttt{p=reject} with reporting.
\end{itemize}
\vfill
\begin{block}{Observation}
\begin{itemize}
	\gitem Weak or inconsistent policy combinations can still let spoofed mail reach the inbox.
\end{itemize}
\end{block}
\end{frame}

\section{Attack Execution}
\begin{frame}[fragile]{Running Espoofer from the Attacker Container}
\begin{minted}{bash}
docker exec attacker-sender-10.9.0.7 \
  /home/seed/espoofer/espoofer \
  -u attacker@example.com \
  -t alice@victim.test \
  -f "Subject: Test" \
  -m "Hello, this is a spoofed mail."
\end{minted}
\vfill
\begin{block}{Observation}
\begin{itemize}
  \gitem The attacker sends a crafted spoofed message toward the victim domain.
  \vspace{0.4cm}
  \gitem Misconfigured validation logic determines whether delivery succeeds.
\end{itemize}
\end{block}
\end{frame}

\section{Results \& Observations}
\begin{frame}{Email Authentication Outcomes}
\centering
\begin{tabular}{lccc}
  \toprule
  \textbf{Test} & \textbf{SPF} & \textbf{DKIM} & \textbf{DMARC} \\
  \midrule
  Baseline (correct config) & Pass & Pass & Pass \\
  Soft-fail SPF             & Pass & Pass & Pass \\
  DKIM selector missing     & Pass & Fail & Pass \\
  DMARC \texttt{p=none}     & Pass & Pass & Pass \\
  \bottomrule
\end{tabular}
\vfill
\begin{itemize}
  \gitem DKIM misconfiguration is the most relevant vector for successful spoofing.
  \vspace{0.4cm}
  \gitem DMARC with \texttt{p=none} gives visibility but no enforcement.
\end{itemize}
\end{frame}

\section{Hardening Recommendations}
\begin{frame}{DNS-Level Fixes}
\begin{itemize}
  \gitem Use \texttt{-all} in SPF to enforce a hard fail.
  \vspace{0.5cm}
  \gitem Publish a valid DKIM selector and rotate keys regularly.
  \vspace{0.5cm}  
  \gitem Publish DMARC with \texttt{p=reject} or \texttt{p=quarantine}.
\end{itemize}
\end{frame}

\begin{frame}{MTA-Level Fixes}
\begin{itemize}
  \gitem Enforce SPF checks on inbound connections.
  \vspace{0.5cm}
  \gitem Reject mail that fails DKIM verification.
  \vspace{0.5cm}
  \gitem Reject mail that violates DMARC policy.
\end{itemize}
\end{frame}
