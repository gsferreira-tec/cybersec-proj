\section{Introduction}
\begin{frame}{Context \& Motivation}
\begin{block}{Why does Email Sender Authentication Matter?}
\begin{itemize}
  \gitem Email remains a primary vector for phishing attacks and weak sender-authentication deployment can create exploits for attackers to appear trustworthy.
  \vspace{0.5cm}
  \gitem SPF, DKIM, and DMARC were designed to mitigate spoofing and throughout this project we study how their misconfiguration can create vulnerabilities.
  \vspace{0.5cm}  
  \gitem We also aim to propose ways to mitigate the impact of the possible attacks that could stem from these vulnerabilities.
\end{itemize}
\end{block}
\end{frame}

\section{Objectives}
\begin{frame}{Objectives}
\begin{block}{What we want to show}
\begin{itemize}
  \gitem Reproduce sender spoofing in a safe, isolated environment that isolates the attack surface from the real internet.
  \vspace{0.5cm}
  \gitem Demonstrate a working SPF-based spoofing path using the attacker container against the victim mail server.
  \vspace{0.5cm}
  \gitem Document the setup clearly so the lab can be extended later for DKIM, DMARC, and hardening tests.
\end{itemize}
\end{block}

\vfill

\begin{block}{\underline{Expected outcome}}
\begin{itemize}
  \gitem A reproducible lab that shows how SPF misconfiguration enables spoofing and how to block it.
  \vspace{0.5cm}
  \gitem Concrete DNS and MTA hardening steps that can be validated in the same environment.
\end{itemize}
\end{block}
\end{frame}

\section{Preliminary solution concept}
\begin{frame}{Preliminary solution concept}
\begin{block}{Lab design}
\begin{itemize}
  \gitem The lab uses a DNS server, a victim mail server, and an attacker container.
  \gitem DNS is configured so SPF-related misconfiguration allows spoofed mail to pass.
  \gitem The current stage focuses on the SPF scenario that already succeeds; DKIM and DMARC remain for later validation.
\end{itemize}
\end{block}

\section{Development methodology and key tools}
\begin{frame}[fragile]{Development methodology and key tools}
\begin{block}{Implementation approach}
\begin{itemize}
  \gitem Docker Compose is used to orchestrate the lab environment.
  \gitem BIND provides the DNS layer, docker-mailserver hosts the victim mail system, and Espoofer runs from the attacker container.
  \gitem Logs and message delivery behavior are used to check whether the spoofed SPF case succeeds.
\end{itemize}
\end{block}
